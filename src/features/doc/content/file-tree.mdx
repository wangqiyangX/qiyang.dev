## Features

- Uses a compound component API for explicit folder and file composition.
- Distinguishes folders from files with expandable folder rows.
- Resolves file icons from extensions and known filenames with `tech-stack-icons`.
- Supports icon overrides for custom file or folder visuals.

## Usage

```tsx
import { FileTree } from "@/components/file-tree"
```

```tsx
<FileTree>
  <FileTree.Folder name="src">
    <FileTree.Folder name="components">
      <FileTree.File name="file-tree.tsx" />
      <FileTree.File name="button.tsx" />
    </FileTree.Folder>
    <FileTree.File name="page.tsx" />
  </FileTree.Folder>
  <FileTree.File name="package.json" />
</FileTree>
```

## Examples

### Controlled Folder

```tsx
const [open, setOpen] = React.useState(true)

<FileTree>
  <FileTree.Folder name="registry" open={open} onOpenChange={setOpen}>
    <FileTree.File name="file-tree.tsx" />
    <FileTree.File name="file-tree-demo.tsx" />
  </FileTree.Folder>
</FileTree>
```

### Custom Icon

```tsx
import { FileTextIcon } from "lucide-react"

function ReadmeTree() {
  return (
    <FileTree>
      <FileTree.File icon={<FileTextIcon />} name="README" />
    </FileTree>
  )
}
```
